# 树

简单的实例

```python
class Tree:
    """
    一颗简单的文件树, 总是现有父节点, 再创建子节点
    """
    def __init__(self, data, parent=None):
        self.data = data
        self.parent = parent
        self.childrens = None

        if self.parent is not None:
            if self.parent.childrens is None:
                self.parent.childrens = [self]
            else:
                self.parent.childrens.append(self)


    # 判断是否为根节点
    def is_root(self):
        return self.parent is None

    # 判断是否为叶子节点
    def is_leaf(self):
        return self.childrens is None

    # 删除子节点, 删除所有等于 data 的子节点
    def delete_children(self, data):
        if self.childrens is None:
            return 

        count = 0
        for index in range(len(self.childrens)):
            if self.childrens[index].data != data:
                if count != index:
                    self.childrens[count] = self.childrens[index]

                count += 1

        self.childrens = self.childrens[:count]

    # 查询所有子节点
    def get_childrens(self):
        if self.childrens is None:
            raise Exception('no childrens')
        return self.childrens


    # 查询子节点
    def find_children(self, data):
        res = []
        for i in self.childrens:
            if i.data == data:
                res.append(i)

        return res


    # 递归查询子节点
    def find_recursive(self, data):
        res = []

        if self.childrens is None:
            return res

        for i in self.childrens:
            if i.data == data:
                res.append(i)

            res += i.find_recursive(data)

        return res


    def __str__(self):
        if self.parent is None:
            return self.data.__str__()

        return self.parent.__str__() + '/' + self.data.__str__()


    __repr__ = __str__



root = Tree('root')
paa = Tree('paa', parent=root)
pdd = Tree('pdd', parent=root)
aa = Tree('aa', parent=paa)
bb = Tree('bb', parent=paa)
cc = Tree('cc', parent=paa)
ee = Tree('aa', parent=aa)
```
